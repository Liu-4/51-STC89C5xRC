void DigitalTube_DispalySingle(u8 position,u8 num_code) //定义位置和数据
{
    // 位选 P15 P14 P13
    position <<=3; //左移三位
    P1 &= 0xC7;     //和11100011 与运算，保留10 11 12 16 17 18脚的数据
    P1 |= position; //讲左移三位后的数据填入
    // 段选:P0
    P0 = num_code; //直接调用数据

}
